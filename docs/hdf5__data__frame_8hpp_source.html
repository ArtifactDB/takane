73inline void validate_row_names(const H5::Group& handle, hsize_t num_rows) try {
74 if (!handle.exists("row_names") || handle.childObjType("row_names") != H5O_TYPE_DATASET) {
75 throw std::runtime_error("expected a 'row_names' dataset when row names are present");
76 }
77 auto rnhandle = handle.openDataSet("row_names");
78 if (rnhandle.getTypeClass() != H5T_STRING) {
79 throw std::runtime_error("expected 'row_names' to be a string dataset");
80 }
81 if (ritsuko::hdf5::get_1d_length(rnhandle.getSpace(), false) != num_rows) {
82 throw std::runtime_error("expected 'row_names' to have length equal to the number of rows");
83 }
84} catch (std::exception& e) {
85 throw std::runtime_error("failed to validate the row names for '" + ritsuko::hdf5::get_name(handle) + "'; " + std::string(e.what()));

286inline void validate(const H5::H5File& handle, const Parameters& params) {
287 if (!handle.exists(params.group) || handle.childObjType(params.group) != H5O_TYPE_GROUP) {
288 throw std::runtime_error("expected a '" + params.group + "' group");
289 }
290 auto ghandle = handle.openGroup(params.group);
291
292 if (params.has_row_names) {
293 validate_row_names(ghandle, params.num_rows);
294 }
295 validate_column_names(ghandle, params);
296
297 if (!ghandle.exists("data") || ghandle.childObjType("data") != H5O_TYPE_GROUP) {
298 throw std::runtime_error("expected a 'data' group");
299 }
300 auto dhandle = ghandle.openGroup("data");
301
302 const auto& columns = *(params.columns);
303 size_t NC = columns.size();
304 hsize_t found = 0;
305 for (size_t c = 0; c < NC; ++c) {
306 const auto& curcol = columns[c];
307
308 std::string dset_name = std::to_string(c);
309 if (!ghandle.exists(dset_name)) {
310 if (curcol.type == data_frame::ColumnType::OTHER) {
311 continue;
312 }
313 }
314
315 validate_column(dhandle, dset_name, curcol, params);
316 ++found;
317 }
318
319 if (found != dhandle.getNumObjs()) {
320 throw std::runtime_error("more objects present in the 'data' group than expected");
321 }
322}
