96 if (!ghandle.exists("column_names") || ghandle.childObjType("column_names") != H5O_TYPE_DATASET) {
97 throw std::runtime_error("expected a 'column_names' dataset");
98 }
99
100 auto cnhandle = ghandle.openDataSet("column_names");
101 if (cnhandle.getTypeClass() != H5T_STRING) {
102 throw std::runtime_error("expected 'column_names' to be a string dataset");
103 }
104
105 const auto& columns = *(params.columns);
106 size_t num_cols = ritsuko::hdf5::get_1d_length(cnhandle.getSpace(), false);
107 if (num_cols != columns.size()) {
108 throw std::runtime_error("length of 'column_names' should equal the expected number of columns");
109 }
110
111 ritsuko::hdf5::load_1d_string_dataset(
112 cnhandle,
113 num_cols,

116 const auto& expected = columns[i].name;
117 if (l != expected.size() || strncmp(expected.c_str(), p, l)) {
118 throw std::runtime_error("expected name '" + expected + "' but got '" + std::string(p, p + l) + "' for column " + std::to_string(i));
119 }
120 }
121 );
122
123 std::unordered_set<std::string> column_names;
124 for (const auto& col : columns) {
125 if (column_names.find(col.name) != column_names.end()) {
126 throw std::runtime_error("duplicated column name '" + col.name + "'");
127 }
128 column_names.insert(col.name);
129 }
130} catch (std::exception& e) {
131 throw std::runtime_error("failed to validate the column names for '" + ritsuko::hdf5::get_name(ghandle) + "'; " + std::string(e.what()));

506inline void validate(const H5::H5File& handle, const Parameters& params) {
507 if (!handle.exists(params.group) || handle.childObjType(params.group) != H5O_TYPE_GROUP) {
508 throw std::runtime_error("expected a '" + params.group + "' group");
509 }
510 auto ghandle = handle.openGroup(params.group);
511
512 // Inspecting the columns.
513 ritsuko::Version version;
514 if (ghandle.attrExists("version")) {
515 auto vstring = ritsuko::hdf5::load_scalar_string_attribute(ghandle, "version");
516 version = ritsuko::parse_version_string(vstring.c_str(), vstring.size(), /* skip_patch = */ true);
517 if (version.major != 1) {
518 throw std::runtime_error("unsupported version '" + vstring + "' for the '" + params.group + "' group");
519 }
520 }
521
522 // Checking the number of rows.
523 if (version.major > 0) {
524 auto attr = ritsuko::hdf5::get_scalar_attribute(ghandle, "num_rows");
525 if (ritsuko::hdf5::exceeds_integer_limit(attr, 64, false)) {
526 throw std::runtime_error("'num_rows' attribute on '" + params.group + "' should have a datatype that fits in a 64-bit unsigned integer");
527 }
528 uint64_t nrows = 0;
529 attr.read(H5::PredType::NATIVE_UINT64, &nrows);
530 if (nrows != params.num_rows) {
531 throw std::runtime_error("inconsistent number of rows in '" + params.group + "' (expected " + std::to_string(params.num_rows) + ", got " + std::to_string(nrows) + ")");
532 }
533 }
534
535 // Checking row and column names.
536 if (params.has_row_names) {
537 validate_row_names(ghandle, params.num_rows);
538 }
539 validate_column_names(ghandle, params);
540
541 // Finally iterating through the columns.
542 if (!ghandle.exists("data") || ghandle.childObjType("data") != H5O_TYPE_GROUP) {
543 throw std::runtime_error("expected a '" + params.group + "/data' group");
544 }
545 auto dhandle = ghandle.openGroup("data");
546
547 const auto& columns = *(params.columns);
548 size_t NC = columns.size();
549 hsize_t found = 0;
550 for (size_t c = 0; c < NC; ++c) {
551 const auto& curcol = columns[c];
552
553 std::string dset_name = std::to_string(c);
554 if (!dhandle.exists(dset_name)) {
555 if (curcol.type == data_frame::ColumnType::OTHER) {
556 continue;
557 }
558 }
559
560 if (version.major > 0) {
561 validate_column_v3(dhandle, dset_name, curcol, params);
562 } else {
563 validate_column_v1_v2(dhandle, dset_name, curcol, params);
564 }
565
566 ++found;
567 }
568
569 if (found != dhandle.getNumObjs()) {
570 throw std::runtime_error("more objects present in the '" + params.group + "/data' group than expected");
571 }
572}
