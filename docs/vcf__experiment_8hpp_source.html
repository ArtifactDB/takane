33std::pair<size_t, size_t> scan_vcf_dimensions(const std::filesystem::path& path, bool expanded) {
34 internal_files::check_gzip_signature(path);
35 auto reader = internal_other::open_reader<byteme::GzipFileReader>(path);
36 typename std::conditional<parallel_, byteme::PerByteParallel<>, byteme::PerByte<> >::type pb(&reader);
37
38 // Checking the signature.
39 {
40 const std::string expected = "##fileformat=VCFv";
41 const size_t len = expected.size();
42 bool okay = pb.valid();
43
44 for (size_t i = 0; i < len; ++i) {
45 if (!okay) {
46 throw std::runtime_error("incomplete VCF file signature");
47 }
48 if (pb.get() != expected[i]) {
49 throw std::runtime_error("incorrect VCF file signature");
50 }
51 okay = pb.advance();
52 }
53 }
54
55 // Scanning until we find a line that doesn't start with '##'.
56 while (true) {
57 if (pb.get() == '\n') {
58 bool escape = false;
59 for (int i = 0; i < 2; ++i) {
60 if (!pb.advance()) {
61 throw std::runtime_error("premature end to the VCF file");
62 }
63 if (pb.get() != '#') {
64 escape = true;
65 break;
66 }
67 }
68 if (escape) {
69 break;
70 }
71 }
72 if (!pb.advance()) {
73 throw std::runtime_error("premature end to the VCF file");
74 }
75 }
76
77 // Scanning the header line to count the number of samples.
78 size_t num_samples = 0;
79 {
80 size_t num_indents = 0;
81 while (true) {
82 char current = pb.get();
83 if (current == '\t') {
84 ++num_indents;
85 } else if (current == '\n') {
86 pb.advance(); // skip past the newline.
87 break;
88 }
89 if (!pb.advance()) {
90 throw std::runtime_error("premature end to the VCF file");
91 }
92 }
93
94 if (num_indents < 8) {
95 throw std::runtime_error("expected at least 9 fields in the VCF header line, including 'FORMAT'");
96 }
97 num_samples = num_indents - 8;
98 }
99
100 size_t expected_rows = 0;
101 if (expanded) {
102 while (pb.valid()) {
103 ++expected_rows;
104
105 // Scanning up to the ALT field, which is the 5th one. We need this to
106 // find the number of alternative alleles (and thus the expansion of rows).
107 size_t num_indents = 0;
108 while (true) {
109 char current = pb.get();
110 if (current == '\t') {
111 ++num_indents;
112 if (num_indents == 4) {
113 if (!pb.advance()) { // get past this indent.
114 throw std::runtime_error("premature end of line for VCF record");
115 }
116 break;
117 }
118 } else if (current == '\n') {
119 throw std::runtime_error("premature end of line for VCF record");
120 }
121 if (!pb.advance()) {
122 throw std::runtime_error("premature end of line for VCF record");
123 }
124 }
125
126 // Checking that we don't have any commas if it's expanded.
127 while (true) {
128 char current = pb.get();
129 if (current == ',') {
130 throw std::runtime_error("expected a 1:1 mapping of rows to alternative alleles when 'vcf_experiment.expanded = true'");
131 } else if (current == '\t') {
132 break;
133 } else if (current == '\n') {
134 throw std::runtime_error("premature end of line for VCF record");
135 }
136 if (!pb.advance()) {
137 throw std::runtime_error("premature end of line for VCF record");
138 }
139 }
140
141 // Chomping the rest of the line; we assume all lines are newline-terminated.
142 while (true) {
143 if (pb.get() == '\n') {
144 pb.advance(); // skip past the newline.
145 break;
146 } else {
147 if (!pb.advance()) {
148 throw std::runtime_error("premature end of line for VCF record");
149 }
150 }
151 }
152 }
153
154 } else {
155 if (pb.valid()) {
156 while (true) {
157 if (pb.get() == '\n') {
158 ++expected_rows; // assume all files are newline-terminated.
159 if (!pb.advance()) {
160 break;
161 }
162 } else {
163 if (!pb.advance()) {
164 throw std::runtime_error("premature end of line for VCF record");
165 }
166 }
167 }
168 }
169 }
170
171 return std::make_pair(expected_rows, num_samples);
172}

181inline void validate(const std::filesystem::path& path, const ObjectMetadata& metadata, const Options& options) {
182 const auto& vcfmap = internal_json::extract_typed_object_from_metadata(metadata.other, "vcf_experiment");
183
184 const std::string& vstring = internal_json::extract_string_from_typed_object(vcfmap, "version", "vcf_experiment");
185 auto version = ritsuko::parse_version_string(vstring.c_str(), vstring.size(), /* skip_patch = */ true);
186 if (version.major != 1) {
187 throw std::runtime_error("unsupported version string '" + vstring + "'");
188 }
189
190 // Other bits and pieces of the metadata to check.
191 auto dims = internal_summarized_experiment::extract_dimensions_json(vcfmap, "vcf_experiment");
192
193 bool exp = false;
194 {
195 auto eIt = vcfmap.find("expanded");
196 if (eIt == vcfmap.end()) {
197 throw std::runtime_error("expected a 'vcf_experiment.expanded' property");
198 }
199 const auto& val = eIt->second;
200 if (val->type() != millijson::BOOLEAN) {
201 throw std::runtime_error("'vcf_experiment.expanded' property should be a JSON boolean");
202 }
203 exp = reinterpret_cast<const millijson::Boolean*>(val.get())->value;
204 }
205
206 auto ipath = path / "file.vcf.gz";
207 std::pair<size_t, size_t> obs_dims;
208 try {
209 if (options.parallel_reads) {
210 obs_dims = internal::scan_vcf_dimensions<true>(ipath, exp);
211 } else {
212 obs_dims = internal::scan_vcf_dimensions<false>(ipath, exp);
213 }
214 } catch (std::exception& e) {
215 throw std::runtime_error("failed to parse '" + ipath.string() + "'; " + std::string(e.what()));
216 }
217
218 if (obs_dims.first != dims.first) {
219 throw std::runtime_error("reported 'vcf_experiment.dimensions[0]' does not match the number of records in '" + ipath.string() + "'");
220 }
221 if (obs_dims.second != dims.second) {
222 throw std::runtime_error("reported 'vcf_experiment.dimensions[1]' does not match the number of samples in '" + ipath.string() + "'");
223 }
224}
