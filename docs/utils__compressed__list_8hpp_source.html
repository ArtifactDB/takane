28inline hsize_t validate_group(const H5::Group& handle, size_t concatenated_length, hsize_t buffer_size) {
29 auto lhandle = ritsuko::hdf5::open_dataset(handle, "lengths");
30 if (ritsuko::hdf5::exceeds_integer_limit(lhandle, 64, false)) {
31 throw std::runtime_error("expected 'lengths' to have a datatype that fits in a 64-bit unsigned integer");
32 }
33
34 size_t len = ritsuko::hdf5::get_1d_length(lhandle.getSpace(), false);
35 ritsuko::hdf5::Stream1dNumericDataset<uint64_t> stream(&lhandle, len, buffer_size);
36 size_t total = 0;
37 for (size_t i = 0; i < len; ++i, stream.next()) {
38 total += stream.get();
39 }
40 if (total != concatenated_length) {
41 throw std::runtime_error("sum of 'lengths' does not equal the height of the concatenated object (got " + std::to_string(total) + ", expected " + std::to_string(concatenated_length) + ")");
42 }
43
44 return len;
45}
