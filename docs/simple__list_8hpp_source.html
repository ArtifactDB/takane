68 throw std::runtime_error("unsupported version string '" + vstring + "'");

71 std::string format = internal::extract_format(metamap);
72
73 auto other_dir = path / "other_contents";
74 int num_external = 0;
75 if (std::filesystem::exists(other_dir)) {

78 throw std::runtime_error("expected 'other_contents' to be a directory");

81 num_external = internal_other::count_directory_entries(other_dir);
82 for (int e = 0; e < num_external; ++e) {
83 auto epath = other_dir / std::to_string(e);
84 if (!std::filesystem::exists(epath)) {
85 throw std::runtime_error("expected an external list object at '" + std::filesystem::relative(epath, path).string() + "'");
86 }
87
88 try {

90 } catch (std::exception& e) {
91 throw std::runtime_error("failed to validate external list object at '" + std::filesystem::relative(epath, path).string() + "'; " + std::string(e.what()));
92 }
93 }
94 }
95
96 if (format == "json.gz") {
97 uzuki2::json::Options opt;
98 opt.parallel = options.parallel_reads;
99 auto gzreader = internal_other::open_reader<byteme::GzipFileReader>(path / "list_contents.json.gz");
100 uzuki2::json::validate(gzreader, num_external, opt);
101 } else if (format == "hdf5") {
