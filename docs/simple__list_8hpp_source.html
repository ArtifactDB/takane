40inline std::string extract_format(const internal_json::JsonObjectMap& map) {
41 auto fIt = map.find("format");
42 if (fIt == map.end()) {
43 return "hdf5";

48 }
49 return reinterpret_cast<millijson::String*>(val.get())->value;

76 if (version.major != 1) {
77 throw std::runtime_error("unsupported version string '" + vstring + "'");
78 }

80 std::string format = internal::extract_format(metamap);
81
82 auto other_dir = path / "other_contents";
83 int num_external = 0;
84 if (std::filesystem::exists(other_dir)) {
85 auto status = std::filesystem::status(other_dir);
86 if (status.type() != std::filesystem::file_type::directory) {
87 throw std::runtime_error("expected 'other_contents' to be a directory");
88 }
89
90 num_external = internal_other::count_directory_entries(other_dir);
91 for (int e = 0; e < num_external; ++e) {
92 std::string curname = std::to_string(e);
93 auto epath = other_dir / curname;
94 if (!std::filesystem::exists(epath)) {
95 throw std::runtime_error("expected an external list object at 'other_contents/" + curname + "'");
96 }
97
98 try {
99 ::takane::validate(epath, options);
100 } catch (std::exception& e) {
101 throw std::runtime_error("failed to validate external list object at 'other_contents/" + curname + "'; " + std::string(e.what()));
102 }
103 }
104 }
105
106 if (format == "json.gz") {
107 uzuki2::json::Options opt;
108 opt.parallel = options.parallel_reads;
109 auto gzreader = internal::open_reader<byteme::GzipFileReader>(path / "list_contents.json.gz");
110 uzuki2::json::validate(gzreader, num_external, opt);
111 } else if (format == "hdf5") {

113 auto ghandle = ritsuko::hdf5::open_group(handle, "simple_list");
114 uzuki2::hdf5::validate(ghandle, num_external);
