50inline SequenceLimits find_sequence_limits(const std::filesystem::path& path, const Options& options) {
51 auto xtype = read_object_type(path);
52 if (xtype != "sequence_information") {
53 throw std::runtime_error("'sequence_information' directory should contain a 'sequence_information' object");
54 }
55 ::takane::validate(path, xtype, options);
56
57 auto fpath = path / "info.h5";
58 H5::H5File handle(fpath, H5F_ACC_RDONLY);
59 auto ghandle = handle.openGroup("sequence_information");
60
61 auto lhandle = ghandle.openDataSet("length");
62 auto num_seq = ritsuko::hdf5::get_1d_length(lhandle.getSpace(), false);
63 ritsuko::hdf5::Stream1dNumericDataset<uint64_t> lstream(&lhandle, num_seq, options.hdf5_buffer_size);
64 auto lmissing = ritsuko::hdf5::open_and_load_optional_numeric_missing_placeholder<uint64_t>(lhandle, "missing-value-placeholder");
65
66 auto chandle = ghandle.openDataSet("circular");
67 ritsuko::hdf5::Stream1dNumericDataset<int32_t> cstream(&chandle, num_seq, options.hdf5_buffer_size);
68 auto cmissing = ritsuko::hdf5::open_and_load_optional_numeric_missing_placeholder<int32_t>(chandle, "missing-value-placeholder");
69
70 SequenceLimits output(num_seq);
71 auto& restricted = output.restricted;
72 auto& seqlen = output.seqlen;
73
74 for (size_t i = 0; i < num_seq; ++i, lstream.next(), cstream.next()) {
75 auto slen = lstream.get();
76 auto circ = cstream.get();
77 seqlen[i] = slen;
78
79 // Skipping restriction if the sequence length is missing OR the sequence is circular.
80 if (lmissing.first && lmissing.second == slen) {
81 continue;
82 }
83 if (circ && !(cmissing.first && cmissing.second == circ)) {
84 continue;
85 }
86
87 restricted[i] = true;
88 }
89
90 return output;
91}

102inline void validate(const std::filesystem::path& path, const Options& options) try {
103 // Figuring out the sequence length constraints.
104 auto limits = internal::find_sequence_limits(path / "sequence_information", options);
105 const auto& restricted = limits.restricted;
106 const auto& seqlen = limits.seqlen;
107 size_t num_sequences = restricted.size();
108
109 // Handling the preamble.
110 auto handle = ritsuko::hdf5::open_file(path / "ranges.h5");
111 auto ghandle = ritsuko::hdf5::open_group(handle, "genomic_ranges");
112
113 auto vstring = ritsuko::hdf5::open_and_load_scalar_string_attribute(ghandle, "version");
114 auto version = ritsuko::parse_version_string(vstring.c_str(), vstring.size(), /* skip_patch = */ true);
115 if (version.major != 1) {
116 throw std::runtime_error("unsupported version string '" + vstring + "'");
117 }
118
119 // Now loading all three components.
120 auto id_handle = ritsuko::hdf5::open_dataset(ghandle, "sequence");
121 auto num_ranges = ritsuko::hdf5::get_1d_length(id_handle, false);
122 if (ritsuko::hdf5::exceeds_integer_limit(id_handle, 64, false)) {
123 throw std::runtime_error("expected 'sequence' to have a datatype that fits into a 64-bit unsigned integer");
124 }
125 ritsuko::hdf5::Stream1dNumericDataset<uint64_t> id_stream(&id_handle, num_ranges, options.hdf5_buffer_size);
126
127 auto start_handle = ritsuko::hdf5::open_dataset(ghandle, "start");
128 if (num_ranges != ritsuko::hdf5::get_1d_length(start_handle, false)) {
129 throw std::runtime_error("'start' and 'sequence' should have the same length");
130 }
131 if (ritsuko::hdf5::exceeds_integer_limit(start_handle, 64, true)) {
132 throw std::runtime_error("expected 'start' to have a datatype that fits into a 64-bit signed integer");
133 }
134 ritsuko::hdf5::Stream1dNumericDataset<int64_t> start_stream(&start_handle, num_ranges, options.hdf5_buffer_size);
135
136 auto width_handle = ritsuko::hdf5::open_dataset(ghandle, "width");
137 if (num_ranges != ritsuko::hdf5::get_1d_length(width_handle, false)) {
138 throw std::runtime_error("'width' and 'sequence' should have the same length");
139 }
140 if (ritsuko::hdf5::exceeds_integer_limit(width_handle, 64, false)) {
141 throw std::runtime_error("expected 'width' to have a datatype that fits into a 64-bit unsigned integer");
142 }
143 ritsuko::hdf5::Stream1dNumericDataset<uint64_t> width_stream(&width_handle, num_ranges, options.hdf5_buffer_size);
144
145 constexpr uint64_t end_limit = std::numeric_limits<int64_t>::max();
146 for (size_t i = 0; i < num_ranges; ++i, id_stream.next(), start_stream.next(), width_stream.next()) {
147 auto id = id_stream.get();
148 if (id >= num_sequences) {
149 throw std::runtime_error("'sequence' must be less than the number of sequences (got " + std::to_string(id) + ")");
150 }
151
152 auto start = start_stream.get();
153 auto width = width_stream.get();
154
155 if (restricted[id]) {
156 if (start < 1) {
157 throw std::runtime_error("non-positive start position (" + std::to_string(start) + ") for non-circular sequence");
158 }
159
160 auto spos = static_cast<uint64_t>(start);
161 auto limit = seqlen[id];
162 if (spos > limit) {
163 throw std::runtime_error("start position beyond sequence length (" + std::to_string(start) + " > " + std::to_string(limit) + ") for non-circular sequence");
164 }
165
166 // The LHS should not overflow as 'spos >= 1' so 'limit - spos + 1' should still be no greater than 'limit'.
167 if (limit - spos + 1 < width) {
168 throw std::runtime_error("end position beyond sequence length (" +
169 std::to_string(start) + " + " + std::to_string(width) + " > " + std::to_string(limit) +
170 ") for non-circular sequence");
171 }
172 }
173
174 bool exceeded = false;
175 if (start > 0) {
176 // 'end_limit - start' is always non-negative as 'end_limit' is the largest value of an int64_t and 'start' is also int64_t.
177 exceeded = (end_limit - static_cast<uint64_t>(start) < width);
178 } else {
179 // 'end_limit - start' will not overflow a uint64_t, because 'end_limit' is the largest value of an int64_t and 'start' as also 'int64_t'.
180 exceeded = (end_limit + static_cast<uint64_t>(-start) < width);
181 }
182 if (exceeded) {
183 throw std::runtime_error("end position beyond the range of a 64-bit integer (" + std::to_string(start) + " + " + std::to_string(width) + ")");
184 }
185 }
186
187 {
188 auto strand_handle = ritsuko::hdf5::open_dataset(ghandle, "strand");
189 if (num_ranges != ritsuko::hdf5::get_1d_length(strand_handle, false)) {
190 throw std::runtime_error("'strand' and 'sequence' should have the same length");
191 }
192 if (ritsuko::hdf5::exceeds_integer_limit(strand_handle, 32, true)) {
193 throw std::runtime_error("expected 'strand' to have a datatype that fits into a 32-bit signed integer");
194 }
195
196 ritsuko::hdf5::Stream1dNumericDataset<int32_t> strand_stream(&strand_handle, num_ranges, options.hdf5_buffer_size);
197 for (hsize_t i = 0; i < num_ranges; ++i, strand_stream.next()) {
198 auto x = strand_stream.get();
199 if (x < -1 || x > 1) {
200 throw std::runtime_error("values of 'strand' should be one of 0, -1, or 1 (got " + std::to_string(x) + ")");
201 }
202 }
203 }
204
205 internal_other::validate_mcols(path, "range_annotations", num_ranges, options);
206 internal_other::validate_metadata(path, "other_annotations", options);
207
208 internal_string::validate_names(ghandle, "name", num_ranges, options.hdf5_buffer_size);
209
210} catch (std::exception& e) {
211 throw std::runtime_error("failed to validate 'genomic_ranges' object at '" + path.string() + "'; " + std::string(e.what()));
212}
