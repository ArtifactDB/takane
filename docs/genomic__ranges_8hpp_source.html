118template<class ParseCommand>
119void validate_base(
120 ParseCommand parse,
121 size_t num_ranges,
122 bool has_names,
123 const std::unordered_set<std::string>& seqnames,
124 const Options& options)
125{
126 comservatory::Contents contents;
127 if (has_names) {
128 contents.fields.emplace_back(new NamesField);
129 }
130
131 {
132 auto ptr = new SeqnamesField;
133 ptr->all_seqnames = &seqnames;
134 contents.fields.emplace_back(ptr);
135 }
136
137 {
138 auto sptr = new StartField;
139 contents.fields.emplace_back(sptr);
140 auto eptr = new EndField;
141 eptr->start = sptr;
142 contents.fields.emplace_back(eptr);
143 }
144
145 contents.fields.emplace_back(new StrandField);
146
147 comservatory::ReadOptions opt;
148 opt.parallel = options.parallel;
149 parse(contents, opt);
150 if (contents.num_records() != num_ranges) {
151 throw std::runtime_error("number of records in the CSV file does not match the expected number of ranges");
152 }
153
154 if (contents.names[0 + has_names] != "seqnames") {
155 throw std::runtime_error("expected the first (non-name) column to be 'seqnames'");
156 }
157 if (contents.names[1 + has_names] != "start") {
158 throw std::runtime_error("expected the second (non-name) column to be 'start'");
159 }
160 if (contents.names[2 + has_names] != "end") {
161 throw std::runtime_error("expected the third (non-name) column to be 'end'");
162 }
163 if (contents.names[3 + has_names] != "strand") {
164 throw std::runtime_error("expected the fourth (non-name) column to be 'strand'");
165 }
