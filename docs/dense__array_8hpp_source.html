34inline bool is_transposed(const H5::Group& ghandle) {
35 if (!ghandle.attrExists("transposed")) {
36 return false;
37 }
38
39 auto attr = ghandle.openAttribute("transposed");
40 if (!ritsuko::hdf5::is_scalar(attr)) {
41 throw std::runtime_error("expected 'transposed' attribute to be a scalar");
42 }
43 if (ritsuko::hdf5::exceeds_integer_limit(attr, 32, true)) {
44 throw std::runtime_error("expected 'transposed' attribute to have a datatype that fits in a 32-bit signed integer");
45 }
46
47 return ritsuko::hdf5::load_scalar_numeric_attribute<int32_t>(attr) != 0;
48}

50inline void validate_string_contents(const H5::DataSet& dhandle, const std::vector<hsize_t>& data_extent, hsize_t buffer_size) {
51 auto stype = dhandle.getDataType();
52 if (!stype.isVariableStr()) {
53 return;
54 }
55
56 for (auto ex : data_extent) {
57 if (ex == 0) {
58 return;
59 }
60 }
61
62 hsize_t ndims = data_extent.size();
63 std::vector<hsize_t> chunk_extent(ndims, 1);
64 auto cplist = dhandle.getCreatePlist();
65 if (cplist.getLayout() == H5D_CHUNKED) {
66 cplist.getChunk(chunk_extent.size(), chunk_extent.data());
67 }
68
69 // Scaling up the block size as much as possible. We start from the
70 // fastest-changing dimension (i.e., the last one in HDF5) and increase it,
71 // and then we move onto the next-fastest dimension, and so on until the
72 // buffer size is exhausted.
73 auto block_extent = chunk_extent;
74 hsize_t block_size = 1;
75 for (hsize_t d = 0; d < ndims; ++d) {
76 block_extent[d] = std::min(block_extent[d], data_extent[d]); // should be a no-op, but we do this just in case.
77 block_size *= block_extent[d];
78 }
79
80 for (hsize_t i = ndims; i > 0; --i) {
81 int multiple = buffer_size / block_size;
82 if (multiple <= 1) {
83 break;
84 }
85 auto d = i - 1;
86 block_size /= block_extent[d];
87 block_extent[d] = std::min(data_extent[d], block_extent[d] * multiple);
88 block_size *= block_extent[d];
89 }
90
91 // Now iterating through the array.
92 std::vector<hsize_t> starts(ndims), counts(block_extent.begin(), block_extent.end());
93 std::vector<char*> buffer(block_size);
94 hsize_t buffer_length = block_size;
95
96 H5::DataSpace mspace, dspace(ndims, data_extent.data());
97 bool finished = false;
98
99 while (!finished) {
100 buffer.resize(buffer_length);
101 dspace.selectHyperslab(H5S_SELECT_SET, counts.data(), starts.data());
102 mspace.setExtentSimple(ndims, counts.data());
103
104 [[maybe_unused]] ritsuko::hdf5::VariableStringCleaner stream(stype.getId(), mspace.getId(), buffer.data());
105 dhandle.read(buffer.data(), stype, mspace, dspace);
106 for (auto x : buffer) {
107 if (x == NULL) {
108 throw std::runtime_error("detected NULL pointer in a variable-length string dataset");
109 }
110 }
111
112 // Attempting a shift from the last dimension as this is the fastest-changing.
113 for (hsize_t i = ndims; i > 0; --i) {
114 auto d = i - 1;
115 starts[d] += block_extent[d];
116 if (starts[d] >= data_extent[d]) {
117 if (d == 0) {
118 finished = true;
119 } else {
120 starts[d] = 0;
121 buffer_length /= counts[d];
122 counts[d] = std::min(data_extent[d], block_extent[d]);
123 buffer_length *= counts[d];
124 }
125 } else {
126 buffer_length /= counts[d];
127 counts[d] = std::min(data_extent[d] - starts[d], block_extent[d]);
128 buffer_length *= counts[d];
129 break;
130 }
131 }
132 }
133}

145inline void validate(const std::filesystem::path& path, const ObjectMetadata& metadata, const Options& options) {
146 auto vstring = internal_json::extract_version_for_type(metadata.other, "dense_array");
147 auto version = ritsuko::parse_version_string(vstring.c_str(), vstring.size(), /* skip_patch = */ true);
148 if (version.major != 1) {
149 throw std::runtime_error("unsupported version '" + vstring + "'");
150 }
151
152 auto handle = ritsuko::hdf5::open_file(path / "array.h5");
153 auto ghandle = ritsuko::hdf5::open_group(handle, "dense_array");
154 internal::is_transposed(ghandle); // just a check, not used here.
155 auto dhandle = ritsuko::hdf5::open_dataset(ghandle, "data");
156
157 auto dspace = dhandle.getSpace();
158 size_t ndims = dspace.getSimpleExtentNdims();
159 if (ndims == 0) {
160 throw std::runtime_error("expected 'data' array to have at least one dimension");
161 }
162 std::vector<hsize_t> extents(ndims);
163 dspace.getSimpleExtentDims(extents.data());
164
165 auto type = ritsuko::hdf5::open_and_load_scalar_string_attribute(ghandle, "type");
166 if (type == "integer") {
167 if (ritsuko::hdf5::exceeds_integer_limit(dhandle, 32, true)) {
168 throw std::runtime_error("expected integer array to have a datatype that fits into a 32-bit signed integer");
169 }
170 } else if (type == "boolean") {
171 if (ritsuko::hdf5::exceeds_integer_limit(dhandle, 32, true)) {
172 throw std::runtime_error("expected boolean array to have a datatype that fits into a 32-bit signed integer");
173 }
174 } else if (type == "number") {
175 if (ritsuko::hdf5::exceeds_float_limit(dhandle, 64)) {
176 throw std::runtime_error("expected number array to have a datatype that fits into a 64-bit float");
177 }
178 } else if (type == "string") {
179 if (dhandle.getTypeClass() != H5T_STRING) {
180 throw std::runtime_error("expected string array to have a string datatype class");
181 }
182 internal::validate_string_contents(dhandle, extents, options.hdf5_buffer_size);
183 } else {
184 throw std::runtime_error("unknown array type '" + type + "'");
185 }
186
187 if (dhandle.attrExists("missing-value-placeholder")) {
188 auto attr = dhandle.openAttribute("missing-value-placeholder");
189 ritsuko::hdf5::check_missing_placeholder_attribute(dhandle, attr);
190 }
191
192 if (ghandle.exists("names")) {
193 internal_array::check_dimnames(ghandle, "names", extents, options);
194 }
195}
