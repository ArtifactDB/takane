34inline bool is_transposed(const H5::Group& ghandle) {
35 if (!ghandle.attrExists("transposed")) {
36 return false;
37 }
38
39 auto attr = ghandle.openAttribute("transposed");
40 if (!ritsuko::hdf5::is_scalar(attr)) {
41 throw std::runtime_error("expected 'transposed' attribute to be a scalar");
42 }
43 if (ritsuko::hdf5::exceeds_integer_limit(attr, 32, true)) {
44 throw std::runtime_error("expected 'transposed' attribute to have a datatype that fits in a 32-bit signed integer");
45 }
46
47 return ritsuko::hdf5::load_scalar_numeric_attribute<int32_t>(attr) != 0;
48}

60inline void validate(const std::filesystem::path& path, const ObjectMetadata& metadata, Options& options) {
61 const std::string type_name = "dense_array"; // use a separate variable to avoid dangling reference warnings from GCC.
62 const auto& vstring = internal_json::extract_version_for_type(metadata.other, type_name);
63 auto version = ritsuko::parse_version_string(vstring.c_str(), vstring.size(), /* skip_patch = */ true);
64 if (version.major != 1) {
65 throw std::runtime_error("unsupported version '" + vstring + "'");
66 }
67
68 auto handle = ritsuko::hdf5::open_file(path / "array.h5");
69 auto ghandle = ritsuko::hdf5::open_group(handle, "dense_array");
70 internal::is_transposed(ghandle); // just a check, not used here.
71 auto dhandle = ritsuko::hdf5::open_dataset(ghandle, "data");
72
73 auto dspace = dhandle.getSpace();
74 size_t ndims = dspace.getSimpleExtentNdims();
75 if (ndims == 0) {
76 throw std::runtime_error("expected 'data' array to have at least one dimension");
77 }
78 std::vector<hsize_t> extents(ndims);
79 dspace.getSimpleExtentDims(extents.data());
80
81 auto type = ritsuko::hdf5::open_and_load_scalar_string_attribute(ghandle, "type");
82 if (type == "integer") {
83 if (ritsuko::hdf5::exceeds_integer_limit(dhandle, 32, true)) {
84 throw std::runtime_error("expected integer array to have a datatype that fits into a 32-bit signed integer");
85 }
86 } else if (type == "boolean") {
87 if (ritsuko::hdf5::exceeds_integer_limit(dhandle, 32, true)) {
88 throw std::runtime_error("expected boolean array to have a datatype that fits into a 32-bit signed integer");
89 }
90 } else if (type == "number") {
91 if (ritsuko::hdf5::exceeds_float_limit(dhandle, 64)) {
92 throw std::runtime_error("expected number array to have a datatype that fits into a 64-bit float");
93 }
94 } else if (type == "string") {
95 if (!ritsuko::hdf5::is_utf8_string(dhandle)) {
96 throw std::runtime_error("expected string array to have a datatype that can be represented by a UTF-8 encoded string");
97 }
98 ritsuko::hdf5::validate_nd_string_dataset(dhandle, extents, options.hdf5_buffer_size);
99 } else {
100 throw std::runtime_error("unknown array type '" + type + "'");
101 }
102
103 if (dhandle.attrExists("missing-value-placeholder")) {
104 auto attr = dhandle.openAttribute("missing-value-placeholder");
105 ritsuko::hdf5::check_missing_placeholder_attribute(dhandle, attr);
106 }
107
108 if (ghandle.exists("names")) {
109 internal_array::check_dimnames(ghandle, "names", extents, options);
110 }
