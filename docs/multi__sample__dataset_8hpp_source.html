36inline void validate(const std::filesystem::path& path, const ObjectMetadata& metadata, const Options& options) {
37 const auto& dmap = internal_json::extract_typed_object_from_metadata(metadata.other, "multi_sample_dataset");
38
39 const std::string& vstring = internal_json::extract_string_from_typed_object(dmap, "version", "multi_sample_dataset");
40 auto version = ritsuko::parse_version_string(vstring.c_str(), vstring.size(), /* skip_patch = */ true);
41 if (version.major != 1) {
42 throw std::runtime_error("unsupported version string '" + vstring + "'");
43 }
44
45 // Sample data should exist.
46 auto sd_path = path / "sample_data";
47 auto sdmeta = read_object_metadata(sd_path);
48 if (!satisfies_interface(sdmeta.type, "DATA_FRAME")) {
49 throw std::runtime_error("object in 'sample_data' should satisfy the 'DATA_FRAME' interface");
50 }
51 try {
52 ::takane::validate(sd_path, sdmeta, options);
53 } catch (std::exception& e) {
54 throw std::runtime_error("failed to validate 'sample_data'; " + std::string(e.what()));
55 }
56 size_t num_samples = ::takane::height(sd_path, sdmeta, options);
57
58 // Checking the experiments.
59 std::vector<size_t> num_columns;
60 auto edir = path / "experiments";
61 if (std::filesystem::exists(edir)) {
62 size_t num_experiments = internal_summarized_experiment::check_names_json(edir);
63 num_columns.reserve(num_experiments);
64
65 for (size_t e = 0; e < num_experiments; ++e) {
66 auto ename = std::to_string(e);
67 auto epath = edir / ename;
68 auto emeta = read_object_metadata(epath);
69
70 if (!satisfies_interface(emeta.type, "SUMMARIZED_EXPERIMENT")) {
71 throw std::runtime_error("object in 'experiments/" + ename + "' should satisfy the 'SUMMARIZED_EXPERIMENT' interface");
72 }
73
74 try {
75 ::takane::validate(epath, emeta, options);
76 } catch (std::exception& e) {
77 throw std::runtime_error("failed to validate 'experiments/" + ename + "'; " + std::string(e.what()));
78 }
79
80 auto dims = ::takane::dimensions(epath, emeta, options);
81 num_columns.push_back(dims[1]);
82 }
83
84 size_t num_dir_obj = internal_other::count_directory_entries(edir);
85 if (num_dir_obj - 1 != num_experiments) { // -1 to account for the names.json file itself.
86 throw std::runtime_error("more objects than expected inside the 'experiments' subdirectory");
87 }
88 }
89
90 // Checking the sample map.
91 if (num_columns.size() > 0) {
92 try {
93 auto handle = ritsuko::hdf5::open_file(path / "sample_map.h5");
94 auto ghandle = ritsuko::hdf5::open_group(handle, "multi_sample_dataset");
95
96 for (size_t e = 0, end = num_columns.size(); e < end; ++e) {
97 auto ename = std::to_string(e);
98 auto dhandle = ritsuko::hdf5::open_dataset(ghandle, ename.c_str());
99 if (ritsuko::hdf5::exceeds_integer_limit(dhandle, 64, false)) {
100 throw std::runtime_error("'multi_sample_dataset/" + ename + "' should have a datatype that fits into a 64-bit unsigned integer");
101 }
102
103 auto len = ritsuko::hdf5::get_1d_length(dhandle.getSpace(), false);
104 if (len != num_columns[e]) {
105 throw std::runtime_error("length of 'multi_sample_dataset/" + ename + "' should equal the number of columns of 'experiments/" + ename + "'");
106 }
107
108 ritsuko::hdf5::Stream1dNumericDataset<uint64_t> stream(&dhandle, len, options.hdf5_buffer_size);
109 for (hsize_t i = 0; i < len; ++i, stream.next()) {
110 auto x = stream.get();
111 if (static_cast<size_t>(x) >= num_samples) {
112 throw std::runtime_error("indices in 'multi_sample_dataset/" + ename + "' should be less than the number of samples");
113 }
114 }
115 }
116
117 if (num_columns.size() != ghandle.getNumObjs()) {
118 throw std::runtime_error("more objects present in the 'multi_sample_dataset' group than expected");
119 }
120 } catch (std::exception& e) {
121 throw std::runtime_error("failed to validate the sample mapping; " + std::string(e.what()));
122 }
123 }
124
125 internal_other::validate_metadata(path, "other_data", options);
126}
