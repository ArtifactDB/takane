76inline std::vector<size_t> dimensions(const std::filesystem::path& path, const ObjectMetadata& metadata, const Options& options) {
77 if (!std::filesystem::exists(path) || std::filesystem::status(path).type() != std::filesystem::file_type::directory) {
78 throw std::runtime_error("expected '" + path.string() + "' to be a directory");
79 }
80
81 auto vrIt = dimensions_registry.find(metadata.type);
82 if (vrIt == dimensions_registry.end()) {
83 throw std::runtime_error("no registered 'dimensions' function for object type '" + metadata.type + "' at '" + path.string() + "'");
84 }
85
86 return (vrIt->second)(path, metadata, options);
87}
