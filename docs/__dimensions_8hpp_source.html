67inline std::vector<size_t> dimensions(const std::filesystem::path& path, const ObjectMetadata& metadata, const Options& options) {
68 if (!std::filesystem::exists(path) || std::filesystem::status(path).type() != std::filesystem::file_type::directory) {
69 throw std::runtime_error("expected '" + path.string() + "' to be a directory");
70 }
71
72 auto vrIt = dimensions_registry.find(metadata.type);
73 if (vrIt == dimensions_registry.end()) {
74 throw std::runtime_error("no registered 'dimensions' function for object type '" + metadata.type + "' at '" + path.string() + "'");
75 }
76
77 return (vrIt->second)(path, metadata, options);
78}
