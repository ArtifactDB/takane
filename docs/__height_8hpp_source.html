88inline size_t height(const std::filesystem::path& path, const ObjectMetadata& metadata, Options& options) {
89 auto cIt = options.custom_height.find(metadata.type);
90 if (cIt != options.custom_height.end()) {
91 return (cIt->second)(path, metadata, options);

94 static const auto height_registry = internal_height::default_registry();

96 if (vrIt == height_registry.end()) {
97 throw std::runtime_error("no registered 'height' function for object type '" + metadata.type + "' at '" + path.string() + "'");
