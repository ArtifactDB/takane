86inline size_t height(const std::filesystem::path& path, const ObjectMetadata& metadata, const Options& options) {
87 if (!std::filesystem::exists(path) || std::filesystem::status(path).type() != std::filesystem::file_type::directory) {
88 throw std::runtime_error("expected '" + path.string() + "' to be a directory");
89 }
90
91 auto vrIt = height_registry.find(metadata.type);
92 if (vrIt == height_registry.end()) {
93 throw std::runtime_error("no registered 'height' function for object type '" + metadata.type + "' at '" + path.string() + "'");
94 }
95
96 return (vrIt->second)(path, metadata, options);
97}
