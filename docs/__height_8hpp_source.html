81inline size_t height(const std::filesystem::path& path, const std::string& type, const Options& options) {
82 if (!std::filesystem::exists(path) || std::filesystem::status(path).type() != std::filesystem::file_type::directory) {
83 throw std::runtime_error("expected '" + path.string() + "' to be a directory");
84 }
85
86 auto vrIt = height_registry.find(type);
87 if (vrIt == height_registry.end()) {
88 throw std::runtime_error("no registered 'height' function for object type '" + type + "' at '" + path.string() + "'");
89 }
90
91 return (vrIt->second)(path, options);
92}
