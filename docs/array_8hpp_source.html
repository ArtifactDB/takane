40void check_dimnames(const H5::H5File& handle, const std::string& dimnames_group, const Dimensions_& dimensions) try {
41 if (!handle.exists(dimnames_group) || handle.childObjType(dimnames_group) != H5O_TYPE_GROUP) {
42 throw std::runtime_error("expected a group");
43 }
44 auto nhandle = handle.openGroup(dimnames_group);
45
46 for (size_t i = 0, ndim = dimensions.size(); i < ndim; ++i) {
47 std::string dim_name = std::to_string(i);
48 if (!nhandle.exists(dim_name)) {
49 continue;
50 }
51
52 if (nhandle.childObjType(dim_name) != H5O_TYPE_DATASET) {
53 throw std::runtime_error("expected '" + dim_name + "' to be a dataset");
54 }
55 check_dimnames(nhandle.openDataSet(dim_name), dimensions[i]);
56 }
57} catch (std::exception& e) {
58 throw std::runtime_error("failed to validate dimnames at '" + dimnames_group + "'; " + std::string(e.what()));
