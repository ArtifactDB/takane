72inline void validate_shape(const H5::Group& dhandle, const Parameters& params) try {
73 const auto& dimensions = params.dimensions;
74 if (!dhandle.exists("shape") || dhandle.childObjType("shape") != H5O_TYPE_DATASET) {
75 throw std::runtime_error("expected a dataset");
76 }
77
78 auto shandle = dhandle.openDataSet("shape");
79 if (params.version >= 3) {
80 if (ritsuko::hdf5::exceeds_integer_limit(shandle, 64, true)) {
81 throw std::runtime_error("expected the datatype to be a subset of a 64-bit signed integer");
82 }
83 } else {
84 if (shandle.getTypeClass() != H5T_INTEGER) {
85 throw std::runtime_error("expected an integer dataset");
86 }
87 }
88
89 size_t len = ritsuko::hdf5::get_1d_length(shandle.getSpace(), false);
90 if (len != 2) {
91 throw std::runtime_error("expected the dataset to be of length 2");
92 }
93
94 int64_t dims[2];
95 shandle.read(dims, H5::PredType::NATIVE_INT64);
96 if (dims[0] < 0 || static_cast<size_t>(dims[0]) != params.dimensions[0]) {
97 throw std::runtime_error("mismatch in first entry (expected " + std::to_string(dimensions[0]) + ", got " + std::to_string(dims[0]) + ")");
98 }
99 if (dims[1] < 0 || static_cast<size_t>(dims[1]) != params.dimensions[1]) {
100 throw std::runtime_error("mismatch in second entry (expected " + std::to_string(dimensions[1]) + ", got " + std::to_string(dims[1]) + ")");
101 }
102} catch (std::exception& e) {
103 throw std::runtime_error("failed to validate sparse matrix shape at '" + ritsuko::hdf5::get_name(dhandle) + "/shape'; " + std::string(e.what()));
104}

115 if (params.version >= 3) {

263inline void validate(const H5::H5File& handle, const Parameters& params) {
264 const auto& group = params.group;
265 if (!handle.exists(group) || handle.childObjType(group) != H5O_TYPE_GROUP) {
266 throw std::runtime_error("expected a '" + group + "' group");
267 }
268 auto dhandle = handle.openGroup(group);
269
270 validate_shape(dhandle, params);
271 size_t num_nonzero = validate_data(dhandle, params);
272 std::vector<uint64_t> indptrs = validate_indptrs(dhandle, params.dimensions[1], num_nonzero, params);
273 validate_indices(dhandle, indptrs, params.dimensions[0], params);
