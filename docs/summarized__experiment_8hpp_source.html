39inline void validate(const std::filesystem::path& path, const ObjectMetadata& metadata, const Options& options) {
40 const auto& semap = internal_json::extract_typed_object_from_metadata(metadata.other, "summarized_experiment");
41
42 const std::string& vstring = internal_json::extract_string_from_typed_object(semap, "version", "summarized_experiment");
43 auto version = ritsuko::parse_version_string(vstring.c_str(), vstring.size(), /* skip_patch = */ true);
44 if (version.major != 1) {
45 throw std::runtime_error("unsupported version string '" + vstring + "'");
46 }
47
48 // Validating the dimensions.
49 size_t num_rows = 0, num_cols = 0;
50 {
51 auto dIt = semap.find("dimensions");
52 if (dIt == semap.end()) {
53 throw std::runtime_error("expected a 'dimensions' property");
54 }
55 const auto& dims = dIt->second;
56 if (dims->type() != millijson::ARRAY) {
57 throw std::runtime_error("expected 'dimensions' to be an array");
58 }
59
60 auto dptr = reinterpret_cast<const millijson::Array*>(dims.get());
61 if (dptr->values.size() != 2) {
62 throw std::runtime_error("expected 'dimensions' to be an array of length 2");
63 }
64
65 size_t counter = 0;
66 for (const auto& x : dptr->values) {
67 if (x->type() != millijson::NUMBER) {
68 throw std::runtime_error("expected 'dimensions' to be an array of numbers");
69 }
70 auto val = reinterpret_cast<const millijson::Number*>(x.get())->value;
71 if (val < 0 || std::floor(val) != val) {
72 throw std::runtime_error("expected 'dimensions' to contain non-negative integers");
73 }
74 if (counter == 0) {
75 num_rows = val;
76 } else {
77 num_cols = val;
78 }
79 ++counter;
80 }
81 }
82
83 // Checking the assays. The directory is also allowed to not exist,
84 // in which case we have no assays.
85 auto adir = path / "assays";
86 if (std::filesystem::exists(adir)) {
87 size_t num_assays = internal_summarized_experiment::check_names_json(adir);
88 for (size_t i = 0; i < num_assays; ++i) {
89 auto aname = std::to_string(i);
90 auto apath = adir / aname;
91 auto ameta = read_object_metadata(apath);
92 ::takane::validate(apath, ameta, options);
93
94 auto dims = ::takane::dimensions(apath, ameta, options);
95 if (dims.size() < 2) {
96 throw std::runtime_error("object in 'assays/" + aname + "' should have two or more dimensions");
97 }
98 if (dims[0] != num_rows) {
99 throw std::runtime_error("object in 'assays/" + aname + "' should have the same number of rows as its parent '" + metadata.type + "'");
100 }
101 if (dims[1] != num_cols) {
102 throw std::runtime_error("object in 'assays/" + aname + "' should have the same number of columns as its parent '" + metadata.type + "'");
103 }
104 }
105
106 size_t num_dir_obj = internal_other::count_directory_entries(adir);
107 if (num_dir_obj - 1 != num_assays) { // -1 to account for the names.json file itself.
108 throw std::runtime_error("more objects than expected inside the 'assays' subdirectory");
109 }
110 }
111
112 auto rd_path = path / "row_data";
113 if (std::filesystem::exists(rd_path)) {
114 auto rdmeta = read_object_metadata(rd_path);
115 if (!satisfies_interface(rdmeta.type, "DATA_FRAME")) {
116 throw std::runtime_error("object in 'row_data' should satisfy the 'DATA_FRAME' interface");
117 }
118 ::takane::validate(rd_path, rdmeta, options);
119 if (::takane::height(rd_path, rdmeta, options) != num_rows) {
120 throw std::runtime_error("data frame at 'row_data' should have number of rows equal to that of the '" + metadata.type + "'");
121 }
122 }
123
124 auto cd_path = path / "column_data";
125 if (std::filesystem::exists(cd_path)) {
126 auto cdmeta = read_object_metadata(cd_path);
127 if (!satisfies_interface(cdmeta.type, "DATA_FRAME")) {
128 throw std::runtime_error("object in 'column_data' should satisfy the 'DATA_FRAME' interface");
129 }
130 ::takane::validate(cd_path, cdmeta, options);
131 if (::takane::height(cd_path, cdmeta, options) != num_cols) {
132 throw std::runtime_error("data frame at 'column_data' should have number of rows equal to the number of columns of its parent '" + metadata.type + "'");
133 }
134 }
135
136 internal_other::validate_metadata(path, "other_data", options);
137}
