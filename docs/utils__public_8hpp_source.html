42inline ObjectMetadata reformat_object_metadata(millijson::Base* raw) {
43 if (raw->type() != millijson::OBJECT) {
44 throw std::runtime_error("metadata should be a JSON object");
45 }

47 ObjectMetadata output;
48 output.other = std::move(reinterpret_cast<millijson::Object*>(raw)->values);

50 auto tIt = output.other.find("type");
51 if (tIt == output.other.end()) {

60 output.type = std::move(reinterpret_cast<millijson::String*>(tval.get())->value);
61 output.other.erase(tIt);
