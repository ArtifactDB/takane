198template<class ParseCommand>
199void validate_base(ParseCommand parse, size_t num_rows, bool has_row_names, const std::vector<data_frame::ColumnDetails>& columns, const Options& options) {
200 comservatory::Contents contents;
201 if (has_row_names) {
202 contents.fields.emplace_back(new TakaneRowNamesField);
203 }
204
205 size_t ncol = columns.size();
206 std::unordered_set<std::string> present;
207 for (size_t c = 0; c < ncol; ++c) {
208 const auto& col = columns[c];
209 if (present.find(col.name) != present.end()) {
210 throw std::runtime_error("duplicate column name '" + col.name + "'");
211 }
212 present.insert(col.name);
213
214 if (col.type == data_frame::ColumnType::INTEGER) {
215 contents.fields.emplace_back(new TakaneIntegerField(0, c));
216
217 } else if (col.type == data_frame::ColumnType::NUMBER) {
218 contents.fields.emplace_back(new comservatory::DummyNumberField);
219
220 } else if (col.type == data_frame::ColumnType::STRING) {
221 if (col.format == data_frame::StringFormat::DATE) {
222 contents.fields.emplace_back(new TakaneDateField(0, c));
223 } else if (col.format == data_frame::StringFormat::DATE_TIME) {
224 contents.fields.emplace_back(new TakaneDateTimeField(0, c));
225 } else {
226 contents.fields.emplace_back(new comservatory::DummyStringField);
227 }
228
229 } else if (col.type == data_frame::ColumnType::BOOLEAN) {
230 contents.fields.emplace_back(new comservatory::DummyBooleanField);
231
232 } else if (col.type == data_frame::ColumnType::FACTOR) {
233 if (options.df_version == 1) {
234 contents.fields.emplace_back(new TakaneFactorV1Field(0, c, &(col.factor_levels)));
235 } else {
236 contents.fields.emplace_back(new TakaneFactorV2Field(0, c, col.factor_levels.size()));
237 }
238 } else if (col.type == data_frame::ColumnType::OTHER) {
239 contents.fields.emplace_back(new comservatory::UnknownField); // This can be anything.
240
241 } else {
242 throw std::runtime_error("unknown code for the expected column type");
243 }
244 }
245
246 comservatory::ReadOptions opt;
247 opt.parallel = options.parallel;
248 parse(contents, opt);
249 if (contents.num_records() != num_rows) {
250 throw std::runtime_error("number of records in the CSV file does not match the expected number of rows");
251 }
252
253 for (size_t c = 0; c < ncol; ++c) {
254 const auto& col = columns[c];
255 if (col.name != contents.names[c + has_row_names]) {
256 throw std::runtime_error("observed and expected header names do not match");
257 }
258 }
