28inline std::vector<uint64_t> validate_dimensions(const H5::Group& handle) {
29 auto dhandle = ritsuko::hdf5::open_dataset(handle, "dimensions");
30 if (ritsuko::hdf5::exceeds_integer_limit(dhandle, 64, false)) {
31 throw std::runtime_error("expected 'dimensions' to have a datatype that fits in a 64-bit unsigned integer");
32 }
33 auto len = ritsuko::hdf5::get_1d_length(dhandle, false);
34 std::vector<uint64_t> output(len);
35 dhandle.read(output.data(), H5::PredType::NATIVE_UINT64);
36 return output;
37}

39inline hsize_t validate_lengths(const H5::Group& handle, size_t concatenated_length, hsize_t buffer_size) {
40 auto lhandle = ritsuko::hdf5::open_dataset(handle, "lengths");
41 if (ritsuko::hdf5::exceeds_integer_limit(lhandle, 64, false)) {
42 throw std::runtime_error("expected 'lengths' to have a datatype that fits in a 64-bit unsigned integer");
43 }
44
45 auto len = ritsuko::hdf5::get_1d_length(lhandle.getSpace(), false);
46
47 ritsuko::hdf5::Stream1dNumericDataset<uint64_t> stream(&lhandle, len, buffer_size);
48 size_t total = 0;
49 for (size_t i = 0; i < len; ++i, stream.next()) {
50 total += stream.get();
51 }
52 if (total != concatenated_length) {
53 throw std::runtime_error("sum of 'lengths' does not equal the height of the concatenated object (got " + std::to_string(total) + ", expected " + std::to_string(concatenated_length) + ")");
54 }
55
56 return len;
57}

59inline void validate_sparse_indices(const H5::Group& handle, const std::vector<uint64_t>& dimensions, hsize_t num_lengths, hsize_t buffer_size) {
60 size_t ndims = dimensions.size();
61 std::vector<H5::DataSet> handles;
62 handles.reserve(ndims);
63
64 for (size_t d = 0; d < ndims; ++d) {
65 auto dname = std::to_string(d);
66 auto lhandle = ritsuko::hdf5::open_dataset(handle, dname.c_str());
67 if (ritsuko::hdf5::exceeds_integer_limit(lhandle, 64, false)) {
68 throw std::runtime_error("expected '" + dname + "' to have a datatype that fits in a 64-bit unsigned integer");
69 }
70 auto len = ritsuko::hdf5::get_1d_length(lhandle.getSpace(), false);
71 if (len != num_lengths) {
72 throw std::runtime_error("expected '" + dname + "' to have the same length as 'lengths'");
73 }
74 handles.push_back(lhandle);
75 }
76
77 if (!num_lengths) {
78 return;
79 }
80
81 std::vector<ritsuko::hdf5::Stream1dNumericDataset<uint64_t> > streams;
82 streams.reserve(ndims);
83 std::vector<uint64_t> position;
84 position.reserve(ndims);
85
86 for (size_t d = 0; d < ndims; ++d) {
87 streams.emplace_back(&(handles[d]), num_lengths, buffer_size);
88 position.push_back(streams.back().get());
89 if (position.back() >= dimensions[d]) {
90 throw std::runtime_error("values in 'indices/" + std::to_string(d) + "' should be less than the corresponding dimension extent");
91 }
92 streams.back().next();
93 }
94
95 for (size_t l = 1; l < num_lengths; ++l) {
96 bool greater = false;
97
98 // Going from back to front and checking for increasingness, based on the
99 // expectation that the first dimension is the fastest-changing.
100 for (size_t rd = ndims; rd > 0; --rd) {
101 auto d = rd - 1;
102 auto& dstream = streams[d];
103
104 auto nextv = dstream.get();
105 if (nextv >= dimensions[d]) {
106 throw std::runtime_error("values in 'indices/" + std::to_string(d) + "' should be less than the corresponding dimension extent");
107 }
108 dstream.next();
109
110 auto& oldv = position[d];
111 if (!greater) {
112 if (nextv > oldv) {
113 greater = true;
114 } else if (nextv < oldv) {
115 throw std::runtime_error("coordinates in 'indices' should be strictly increasing");
116 }
117 }
118 oldv = nextv;
119 }
120
121 if (!greater) {
122 throw std::runtime_error("duplicate coordinates in 'indices'");
123 }
124 }
125}

127template<bool satisfies_interface_>
128void validate_directory(const std::filesystem::path& path, const std::string& object_type, const std::string& concatenated_type, const ObjectMetadata& metadata, const Options& options) try {
129 auto vstring = internal_json::extract_version_for_type(metadata.other, object_type);
130 auto version = ritsuko::parse_version_string(vstring.c_str(), vstring.size(), /* skip_patch = */ true);
131 if (version.major != 1) {
132 throw std::runtime_error("unsupported version string '" + vstring + "'");
133 }
134
135 auto catdir = path / "concatenated";
136 auto catmeta = read_object_metadata(catdir);
137 if constexpr(satisfies_interface_) {
138 if (!satisfies_interface(catmeta.type, concatenated_type)) {
139 throw std::runtime_error("'concatenated' should satisfy the '" + concatenated_type + "' interface");
140 }
141 } else {
142 if (catmeta.type != concatenated_type) {
143 throw std::runtime_error("'concatenated' should contain an '" + concatenated_type + "' object");
144 }
145 }
146
147 try {
148 ::takane::validate(catdir, catmeta, options);
149 } catch (std::exception& e) {
150 throw std::runtime_error("failed to validate the 'concatenated' object; " + std::string(e.what()));
151 }
152 size_t catheight = ::takane::height(catdir, catmeta, options);
153
154 auto handle = ritsuko::hdf5::open_file(path / "partitions.h5");
155 auto ghandle = ritsuko::hdf5::open_group(handle, object_type.c_str());
156
157 auto dims = validate_dimensions(ghandle);
158 auto len = validate_lengths(ghandle, catheight, options.hdf5_buffer_size);
159
160 // Support both dense and sparse cases.
161 if (ghandle.exists("indices")) {
162 auto ihandle = ritsuko::hdf5::open_group(ghandle, "indices");
163 validate_sparse_indices(ihandle, dims, len, options.hdf5_buffer_size);
164 } else {
165 size_t prod = !dims.empty(); // prod = 0 if dims is empty, otherwise it starts at 1.
166 for (auto d : dims) {
167 prod *= d;
168 }
169 if (prod != len) {
170 throw std::runtime_error("length of 'lengths' should equal the product of 'dimensions'");
171 }
172 }
173
174 if (ghandle.exists("names")) {
175 internal_array::check_dimnames(ghandle, "names", dims, options);
176 }
177
178} catch (std::exception& e) {
179 throw std::runtime_error("failed to validate a '" + object_type + "' object at '" + path.string() + "'; " + std::string(e.what()));
180}
