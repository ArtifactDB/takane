43inline void validate_row_names(const H5::Group& handle, hsize_t num_rows, const Options& options) try {
44 if (handle.childObjType("row_names") != H5O_TYPE_DATASET) {
45 throw std::runtime_error("expected a 'row_names' dataset when row names are present");
46 }
47
48 auto rnhandle = handle.openDataSet("row_names");
49 if (!ritsuko::hdf5::is_utf8_string(rnhandle)) {
50 throw std::runtime_error("expected a datatype for 'row_names' that can be represented by a UTF-8 encoded string");
51 }
52
53 if (ritsuko::hdf5::get_1d_length(rnhandle.getSpace(), false) != num_rows) {
54 throw std::runtime_error("expected 'row_names' to have length equal to the number of rows");
55 }
56 ritsuko::hdf5::validate_1d_string_dataset(rnhandle, num_rows, options.hdf5_buffer_size);
57} catch (std::exception& e) {
58 throw std::runtime_error("failed to validate the row names for '" + ritsuko::hdf5::get_name(handle) + "'; " + std::string(e.what()));
59}

61inline hsize_t validate_column_names(const H5::Group& ghandle, const Options& options) try {
62 auto cnhandle = ritsuko::hdf5::open_dataset(ghandle, "column_names");
63 if (!ritsuko::hdf5::is_utf8_string(cnhandle)) {
64 throw std::runtime_error("expected a datatype for 'column_names' that can be represented by a UTF-8 encoded string");
65 }
66
67 auto num_cols = ritsuko::hdf5::get_1d_length(cnhandle.getSpace(), false);
68
69 std::unordered_set<std::string> column_names;
70 ritsuko::hdf5::Stream1dStringDataset stream(&cnhandle, num_cols, options.hdf5_buffer_size);
71 for (size_t c = 0; c < num_cols; ++c, stream.next()) {
72 auto x = stream.steal();
73 if (x.empty()) {
74 throw std::runtime_error("column names should not be empty strings");
75 }
76 if (column_names.find(x) != column_names.end()) {
77 throw std::runtime_error("duplicated column name '" + x + "'");
78 }
79 column_names.insert(std::move(x));
80 }
81
82 return num_cols;
83
84} catch (std::exception& e) {
85 throw std::runtime_error("failed to validate the column names for '" + ritsuko::hdf5::get_name(ghandle) + "'; " + std::string(e.what()));
86}

88inline void validate_column(const H5::Group& dhandle, const std::string& dset_name, hsize_t num_rows, const ritsuko::Version& version, const Options& options) try {
89 auto dtype = dhandle.childObjType(dset_name);
90 if (dtype == H5O_TYPE_GROUP) {
91 auto ghandle = dhandle.openGroup(dset_name);
92 auto type = ritsuko::hdf5::open_and_load_scalar_string_attribute(ghandle, "type");
93
94 if (type == "factor") {
95 internal_factor::check_ordered_attribute(ghandle);
96 auto num_levels = internal_factor::validate_factor_levels(ghandle, "levels", options.hdf5_buffer_size);
97 auto num_codes = internal_factor::validate_factor_codes(ghandle, "codes", num_levels, options.hdf5_buffer_size);
98 if (num_codes != num_rows) {
99 throw std::runtime_error("expected column to have length equal to the number of rows");
100 }
101
102 } else if (type == "vls") {
103 if (version.lt(1, 1, 0)) {
104 throw std::runtime_error("unsupported type '" + type + "'");
105 }
106
107 auto phandle = ritsuko::hdf5::vls::open_pointers(ghandle, "pointers", 64, 64);
108 auto vlen = ritsuko::hdf5::get_1d_length(phandle.getSpace(), false);
109 if (vlen != num_rows) {
110 throw std::runtime_error("expected column to have length equal to the number of rows");
111 }
112
113 auto hhandle = ritsuko::hdf5::vls::open_heap(ghandle, "heap");
114 auto hlen = ritsuko::hdf5::get_1d_length(hhandle.getSpace(), false);
115 ritsuko::hdf5::vls::validate_1d_array<uint64_t, uint64_t>(phandle, vlen, hlen, options.hdf5_buffer_size);
116
117 const char* missing_attr_name = "missing-value-placeholder";
118 if (phandle.attrExists(missing_attr_name)) {
119 auto attr = phandle.openAttribute(missing_attr_name);
120 { // TODO: replace the section with check_string_missing_placeholder_attribute()
121 if (!ritsuko::hdf5::is_scalar(attr)) {
122 throw std::runtime_error("expected the '" + ritsuko::hdf5::get_name(attr) + "' attribute to be a scalar");
123 }
124 if (attr.getTypeClass() != H5T_STRING) {
125 throw std::runtime_error("expected the '" + ritsuko::hdf5::get_name(attr) + "' attribute to have the same type class as its dataset");
126 }
127 ritsuko::hdf5::validate_scalar_string_attribute(attr);
128 }
129 }
130
131 } else {
132 throw std::runtime_error("unsupported type '" + type + "'");
133 }
134
135 } else if (dtype == H5O_TYPE_DATASET) {
136 auto xhandle = dhandle.openDataSet(dset_name);
137 if (num_rows != ritsuko::hdf5::get_1d_length(xhandle.getSpace(), false)) {
138 throw std::runtime_error("expected column to have length equal to the number of rows");
139 }
140
141 const char* missing_attr_name = "missing-value-placeholder";
142
143 auto type = ritsuko::hdf5::open_and_load_scalar_string_attribute(xhandle, "type");
144 if (type == "string") {
145 if (!ritsuko::hdf5::is_utf8_string(xhandle)) {
146 throw std::runtime_error("expected a datatype for '" + dset_name + "' that can be represented by a UTF-8 encoded string");
147 }
148 auto missingness = ritsuko::hdf5::open_and_load_optional_string_missing_placeholder(xhandle, missing_attr_name);
149 std::string format = internal_string::fetch_format_attribute(xhandle);
150 internal_string::validate_string_format(xhandle, num_rows, format, missingness.first, missingness.second, options.hdf5_buffer_size);
151
152 } else {
153 if (type == "integer") {
154 if (ritsuko::hdf5::exceeds_integer_limit(xhandle, 32, true)) {
155 throw std::runtime_error("expected integer column to use a datatype that is a subset of a 32-bit signed integer");
156 }
157 } else if (type == "boolean") {
158 if (ritsuko::hdf5::exceeds_integer_limit(xhandle, 32, true)) {
159 throw std::runtime_error("expected boolean column to use a datatype that is a subset of a 32-bit signed integer");
160 }
161 } else if (type == "number") {
162 if (ritsuko::hdf5::exceeds_float_limit(xhandle, 64)) {
163 throw std::runtime_error("expected number column to use a datatype that is a subset of a 64-bit float");
164 }
165 } else {
166 throw std::runtime_error("unknown column type '" + type + "'");
167 }
168
169 if (xhandle.attrExists(missing_attr_name)) {
170 auto ahandle = xhandle.openAttribute(missing_attr_name);
171 ritsuko::hdf5::check_missing_placeholder_attribute(xhandle, ahandle);
172 }
173 }
174
175 } else {
176 throw std::runtime_error("unknown HDF5 object type");
177 }
178
179} catch (std::exception& e) {
180 throw std::runtime_error("failed to validate column at '" + ritsuko::hdf5::get_name(dhandle) + "/" + dset_name + "'; " + std::string(e.what()));
181}
191inline void validate(const std::filesystem::path& path, const ObjectMetadata& metadata, Options& options) {
192 const std::string type_name = "data_frame"; // use a separate variable to avoid dangling reference warnings from GCC.
193 const auto& vstring = internal_json::extract_version_for_type(metadata.other, type_name);
194 auto version = ritsuko::parse_version_string(vstring.c_str(), vstring.size(), /* skip_patch = */ true);
195 if (version.major != 1) {
196 throw std::runtime_error("unsupported version '" + vstring + "'");
197 }
198
199 auto handle = ritsuko::hdf5::open_file(path / "basic_columns.h5");
200 auto ghandle = ritsuko::hdf5::open_group(handle, type_name.c_str());
201
202 // Checking the number of rows.
203 auto attr = ritsuko::hdf5::open_scalar_attribute(ghandle, "row-count");
204 if (ritsuko::hdf5::exceeds_integer_limit(attr, 64, false)) {
205 throw std::runtime_error("'row-count' attribute should have a datatype that fits in a 64-bit unsigned integer");
206 }
207 uint64_t num_rows = 0;
208 attr.read(H5::PredType::NATIVE_UINT64, &num_rows);
209
210 // Checking row and column names.
211 if (ghandle.exists("row_names")) {
212 validate_row_names(ghandle, num_rows, options);
213 }
214 size_t NC = validate_column_names(ghandle, options);
215
216 // Finally iterating through the columns.
217 auto dhandle = ritsuko::hdf5::open_group(ghandle, "data");
218
219 hsize_t num_basic = 0;
220 auto other_dir = path / "other_columns";
221
222 for (size_t c = 0; c < NC; ++c) {
223 std::string dset_name = std::to_string(c);
224
225 if (!dhandle.exists(dset_name)) {
226 auto opath = other_dir / dset_name;
227 auto ometa = read_object_metadata(opath);
228 try {
229 ::takane::validate(opath, ometa, options);
230 } catch (std::exception& e) {
231 throw std::runtime_error("failed to validate 'other' column " + dset_name + "; " + std::string(e.what()));
232 }
233 if (::takane::height(opath, ometa, options) != num_rows) {
234 throw std::runtime_error("height of column " + dset_name + " of class '" + ometa.type + "' is not the same as the number of rows");
235 }
236
237 } else {
238 validate_column(dhandle, dset_name, num_rows, version, options);
239 ++num_basic;
240 }
241 }
242
243 if (std::filesystem::exists(other_dir)) {
244 if (internal_other::count_directory_entries(other_dir) != NC - num_basic) {
245 throw std::runtime_error("more objects than expected inside the 'other_columns' directory");
246 }
247 }
248
249 if (num_basic != dhandle.getNumObjs()) {
250 throw std::runtime_error("more objects present in the 'data_frame/data' group than expected");
251 }
252
253 internal_other::validate_mcols(path, "column_annotations", NC, options);
254 internal_other::validate_metadata(path, "other_annotations", options);
255}
