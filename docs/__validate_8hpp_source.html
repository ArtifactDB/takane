75inline void validate(const std::filesystem::path& path, const std::string& type, const Options& options) {
76 if (!std::filesystem::exists(path) || std::filesystem::status(path).type() != std::filesystem::file_type::directory) {
77 throw std::runtime_error("expected '" + path.string() + "' to be a directory");
78 }
79
80 auto vrIt = validate_registry.find(type);
81 if (vrIt == validate_registry.end()) {
82 throw std::runtime_error("no registered 'validate' function for object type '" + type + "' at '" + path.string() + "'");
83 }
84
85 (vrIt->second)(path, options);
86}
