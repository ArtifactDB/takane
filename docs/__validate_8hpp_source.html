107inline void validate(const std::filesystem::path& path, const ObjectMetadata& metadata, Options& options) {
108 auto cIt = options.custom_validate.find(metadata.type);
109
110 if (cIt != options.custom_validate.end()) {
111 try {
112 (cIt->second)(path, metadata, options);
113 } catch (std::exception& e) {
114 throw std::runtime_error("failed to validate '" + metadata.type + "' object at '" + path.string() + "'; " + std::string(e.what()));
115 }
116
117 } else {
118 static const auto validate_registry = internal_validate::default_registry();
119 auto vrIt = validate_registry.find(metadata.type);
120 if (vrIt == validate_registry.end()) {
121 throw std::runtime_error("no registered 'validate' function for object type '" + metadata.type + "' at '" + path.string() + "'");
122 }
123
124 // Can't easily roll this out, as this is const and the above is not.
125 try {
126 (vrIt->second)(path, metadata, options);
127 } catch (std::exception& e) {
128 throw std::runtime_error("failed to validate '" + metadata.type + "' object at '" + path.string() + "'; " + std::string(e.what()));
129 }
130 }
131
132 if (options.custom_global_validate) {
133 try {
134 options.custom_global_validate(path, metadata, options);
135 } catch (std::exception& e) {
136 throw std::runtime_error("failed additional validation for '" + metadata.type + "' at '" + path.string() + "'; " + std::string(e.what()));
137 }
138 }
139}
